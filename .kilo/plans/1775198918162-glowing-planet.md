# Web UI for Microservices - Implementation Plan

## Overview

Add a web-based control panel to all three services (A, B, C) that allows manual message sending, viewing message history, and monitoring service status. This is a major change that will be developed in the `feat/web-ui` branch.

## Branch Strategy

1. Create new branch: `feat/web-ui` from `main`
2. All changes will be committed to this branch
3. Each service will be updated independently but consistently

## Architecture

### Backend Changes (per service)

Each service needs the following additions:

1. **New API Endpoints**
   - `GET /` - Serve the web UI (HTML page)
   - `GET /api/messages` - Return message history from SQLite database
   - `GET /api/status` - Return service status (port, uptime, connected services)
   - `POST /api/ping` - Manually trigger a ping to another service (JSON body: `{ target: "Service-B" }`)
   - `POST /api/ping/custom` - Send custom message to any service (JSON body: `{ target, message }`)

2. **Static File Serving**
   - Add `public/` directory to each service
   - Serve via `express.static('public')`
   - Contains: `index.html`, `style.css`, `app.js`

3. **State Tracking**
   - Track service start time for uptime calculation
   - Track interval timer references for start/stop control

### Frontend Design (per service)

Single-page application with these sections:

1. **Service Status Panel**
   - Service name and port
   - Uptime counter
   - Auto-ping interval status (running/stopped)
   - Start/Stop interval button

2. **Send Message Panel**
   - Dropdown to select target service
   - Text input for custom message (optional, defaults to "Ping from Service X")
   - "Send Ping" button
   - Response display area

3. **Message History Panel**
   - Table showing last 50 messages
   - Columns: Type (incoming/outgoing), From/To, Message, Timestamp, Status
   - Auto-refresh toggle (every 5 seconds)
   - Manual refresh button

  const { target } = req.body;

  } else if (target === 'Service C') {

  }

app.post('/api/ping/custom', async (req, res) => {
  const { target, message } = req.body;
  // Similar to above but with custom message payload
});

// Control interval
app.post('/api/interval/stop', (req, res) => {
  if (pingInterval) {
    clearInterval(pingInterval);
    pingInterval = null;
    res.json({ success: true, intervalRunning: false });
  } else {
    res.json({ success: false, message: 'Interval already stopped' });
  }
});

app.post('/api/interval/start', (req, res) => {
  if (!pingInterval) {
    pingInterval = setInterval(() => {
      pingServiceB();
      pingServiceC();
    }, 10000);
    res.json({ success: true, intervalRunning: true });
  } else {
    res.json({ success: false, message: 'Interval already running' });
  }
});

#### `public/app.js`

```javascript
const API_BASE = '';

let autoRefreshInterval = null;

async function fetchStatus() {
  const res = await fetch(`${API_BASE}/api/status`);
  const data = await res.json();
  document.getElementById('service-name').textContent = data.service;
  document.getElementById('service-port').textContent = `Port ${data.port}`;
  document.getElementById('uptime').textContent = formatUptime(data.uptime);
  document.getElementById('interval-status').textContent = data.intervalRunning ? 'Running' : 'Stopped';
  document.getElementById('targets').textContent = data.targets.join(', ');

  const select = document.getElementById('target-select');
  select.innerHTML = '';
  data.targets.forEach(t => {
    const opt = document.createElement('option');
    opt.value = t;
    opt.textContent = t;
    select.appendChild(opt);
  });
}

async function fetchMessages() {
  const res = await fetch(`${API_BASE}/api/messages?limit=50`);
  const messages = await res.json();
  const tbody = document.getElementById('history-body');
  tbody.innerHTML = '';
  messages.forEach(msg => {
    const tr = document.createElement('tr');
    const payload = parseJSON(msg.payload);
    const typeClass = msg.direction === 'received' ? 'type-incoming' : 'type-outgoing';
    const statusClass = msg.status === 'success' ? 'status-success' : 'status-error';
    tr.innerHTML = `
      <td class="${typeClass}">${msg.direction === 'received' ? 'IN' : 'OUT'}</td>
      <td>${msg.service}</td>
      <td>${payload?.message || '--'}</td>
      <td class="${statusClass}">${msg.status}</td>
      <td>${msg.created_at}</td>
    `;
    tbody.appendChild(tr);
  });
}

async function sendPing() {
  const target = document.getElementById('target-select').value;
  const customMsg = document.getElementById('custom-message').value;
  const responseEl = document.getElementById('send-response');

  try {
    let res;
    if (customMsg) {
      res = await fetch(`${API_BASE}/api/ping/custom`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ target, message: customMsg })
      });
    } else {
      res = await fetch(`${API_BASE}/api/ping`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ target })
      });
    }
    const data = await res.json();
    if (data.success) {
      responseEl.className = 'success';
      responseEl.textContent = `Ping sent to ${target}`;
    } else {
      responseEl.className = 'error';
      responseEl.textContent = data.error || 'Failed to send';
    }
  } catch (err) {
    responseEl.className = 'error';
    responseEl.textContent = err.message;
  }
  setTimeout(() => { responseEl.style.display = 'none'; }, 3000);
  fetchMessages();
}

async function controlInterval(action) {
  await fetch(`${API_BASE}/api/interval/${action}`, { method: 'POST' });
  fetchStatus();
}

function formatUptime(ms) {
  const seconds = Math.floor(ms / 1000);
  const minutes = Math.floor(seconds / 60);
  const hours = Math.floor(minutes / 60);
  if (hours > 0) return `${hours}h ${minutes % 60}m`;
  if (minutes > 0) return `${minutes}m ${seconds % 60}s`;
  return `${seconds}s`;
}

function parseJSON(str) {
  try { return JSON.parse(str); } catch { return null; }
}

function toggleAutoRefresh(enabled) {
  if (autoRefreshInterval) clearInterval(autoRefreshInterval);
  if (enabled) {
    autoRefreshInterval = setInterval(fetchMessages, 5000);
  }
}

document.getElementById('send-ping').addEventListener('click', sendPing);
document.getElementById('stop-interval').addEventListener('click', () => controlInterval('stop'));
document.getElementById('start-interval').addEventListener('click', () => controlInterval('start'));
document.getElementById('refresh-history').addEventListener('click', fetchMessages);
document.getElementById('auto-refresh').addEventListener('change', (e) => toggleAutoRefresh(e.target.checked));

fetchStatus();
fetchMessages();
```
